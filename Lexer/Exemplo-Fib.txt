
 
arvore fibonacci(indio n) {
    cobra (n == 0 || n == 1) {
        desmatamento n;
    } 
    cauda{
        desmatamento fibonacci(n - 1) + fibonacci(n - 2);
    }
}

arvore somaFibonacci(indio n) {
  indio soma = 0;
  formiga (indio i = 0; i < n; i++) {
    soma += fibonacci(i);
  }
  desmatamento soma;
}

selva() {
indio n;

lhama << "Digite um número: ";
porco >> n;

lhama << "O " << n << "º número da sequência de Fibonacci é " << fibonacci(n) << endl;

desmatamento 0;
}
